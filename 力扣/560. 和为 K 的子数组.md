给你一个整数数组 `nums` 和一个整数 `k` ，请你统计并返回 _该数组中和为 `k` 的子数组的个数_ 。

子数组是数组中元素的连续非空序列。
```
**示例 1：**

**输入：**nums = [1,1,1], k = 2
**输出：**2([0,1], [1,2])

**示例 2：**

**输入：**nums = [1,2,3], k = 3
**输出：**2 ([1,2], [3]
```

```
/**

* @param {number[]} nums

* @param {number} k

* @return {number}

*/

var subarraySum = function(nums, k) {

const map = new Map()

map.set(0, 1) // 初始化，key代表前缀和，value代表前缀和出现的次数

let count = 0 // 子数组和为k的个数

let preFix = 0 // 前缀和

for(const i of nums) {

preFix += i // 计算前缀和

if (map.has(preFix - k)) { // 如果前缀和减去k在map中存在，说明存在一个子数组的和为k

count += map.get(preFix - k) // 将前缀和减去k的次数加到count中

}

if (map.has(preFix)) { // 如果前缀和在map中存在，说明之前出现过这个前缀和，说明存在一个子数组的和为0

map.set(preFix, map.get(preFix) + 1) // 将前缀和的次数加1

} else {

map.set(preFix, 1) // 将前缀和加入map中，次数为1

}

}

console.log(map)

return count // 返回子数组和为k的个数

};

console.log(subarraySum([1,2,3], 3))
```
