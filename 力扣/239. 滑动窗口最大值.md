给你一个整数数组 `nums`，有一个大小为 `k` 的滑动窗口从数组的最左侧移动到数组的最右侧。你只可以看到在滑动窗口内的 `k` 个数字。滑动窗口每次只向右移动一位。

返回 _滑动窗口中的最大值_ 。
```
**输入：**nums = [1,3,-1,-3,5,3,6,7], k = 3
**输出：**[3,3,5,5,6,7]
**解释：**
滑动窗口的位置                最大值
---------------               -----
[1  3  -1] -3  5  3  6  7       **3**
 1 [3  -1  -3] 5  3  6  7       **3**
 1  3 [-1  -3  5] 3  6  7      ** 5**
 1  3  -1 [-3  5  3] 6  7       **5**
 1  3  -1  -3 [5  3  6] 7       **6**
 1  3  -1  -3  5 [3  6  7]      **7**

**示例 2：**

**输入：**nums = [1], k = 1
**输出：**[1]
```

```
/**

* @param {number[]} nums

* @param {number} k

* @return {number[]}

*/

var maxSlidingWindow = function(nums, k) {

let result = []; // 存放结果

let queue = [] // 单调队列，存放当前窗口内的元素，队首为最大值

for(let i = 0; i< nums.length; i++) {

if (queue.length === 0) {

queue.push(nums[i])

} else {

if (i - k >= 0 && nums[i - k] === queue[0]) queue.shift() // 移除滑出窗口的元素

while(queue.length > 0 && nums[i] > queue[queue.length - 1]) { // 移除比当前元素小的元素

queue.pop()

}

queue.push(nums[i]) // 添加当前元素

}

if (i >= k - 1) { // 当窗口形成时，记录当前最大值

result.push(queue[0]) // 队首为当前窗口的最大值

}

}

return result

};

console.log(maxSlidingWindow([1,3,-1,-3,5,3,6,7], 3))
```
